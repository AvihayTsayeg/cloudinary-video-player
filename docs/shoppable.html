        var cld = cloudinary.Cloudinary.new({ cloud_name: 'demo' });

        // Initialize player
        var player = cld.videoPlayer('player');

        // Define source
        var source = {
          info: {
            title: 'Shoppable Video',
            description: 'Browse products shown in the video.',
          },
          shoppable: {
            products : [
                {
                  productName: 'Product 1',
                  publicId: 'ihkanprdxlxcbzp3i7x0',
                  onHover: {
                    action: 'tooltip',
                    args: 'Product 1 tooltip'
                  }
                },
              {
                productName: 'Product 2',
                publicId: 'dqhwndse1fjrdxbmrbvq',
                onHover: {
                  action: 'switch',
                  args: 'ihkanprdxlxcbzp3i7x0'
                }
              },
              {
                productName: 'Product 3',
                publicId: 'wj86zasak3y7vh5j0dhg',
                onHover: {
                  action: 'tooltip',
                  args: 'Product 3 tooltip'
                }
              },
              {
                productName: 'Product 4',
                publicId: 'vaix7e5hbbxxu108a9a3'
              },
              {
                productName: 'Product 5',
                publicId: 'dg0gqxbhm51uu72cyrly'
              }
            ]
          }
        };

        // Play source
        player.source(source);
